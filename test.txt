class Main
{
static void run()
{
int a;
print("Zadejte cislo pro vypocet faktorialu: ");
a = readInt();
int vysl;
if (a < 0) { // nacitani zaporneho cisla nemusite podporovat
print("Faktorial nelze spocitat!\n");
}
else {
vysl = 1;
while (a > 0) {
vysl = vysl * a;
a = a - 1;
}
print("Vysledek je: " + vysl + "\n");
}
}
}
